     27    reg        [DATA_WIDTH-1:0] i_dinb;
     28    reg        [ADDR_WIDTH-1:0] i_addrb;
     29    reg                         i_enb;
     30    reg                         i_clkb;

     82  // Declared the task to generate teh Random stimuls and give it to the random adderss and data  of the port a

     84    task  test_porta(); 
     85     i_dina= $urandom();
     86     i_addra=$urandom_range(0,17);     // Range is Randomized since the no of address loactions are fixed and  depend on the addrss width

     89     i_wea =1'b1;
     90     $display("Data Written into the memory of the adders[=%0d] and data[=%0d] ",i_addra ,i_dina, );
     91     repeat(5) @( posedge (i_clka));
     92     i_ena =1'b1;
     93     i_wea =1'b0;

     95     repeat(5) @( posedge (i_clka));

     97     i_wea =1'b1;
     98     repeat(5) @( posedge (i_clka));
     99     i_ena =1'b0;
     100     i_wea =1'b0;
     101     repeat(5) @( posedge (i_clka));
     102     i_ena =1'b1;
     103     i_wea =1'b0;
     104      repeat(5) @( posedge (i_clka));
     105      i_ena =1'b1;
     106      i_wea =1'b1;
     107  
     108    endtask

     110  // Declared the task to generate teh Random stimuls and give it to the random adderss and data  of the port b

     112     task test_portb();  
     113       i_dinb= $urandom();

     115       repeat(4) @( posedge (i_clkb));
     116       i_enb =1'b1;
     117       i_web =1'b1;
     118       $display("Data Written into the memory of the adders[=%0d] and data[=%0d] ",i_dinb, i_addrb );
     119       repeat(5) @( posedge (i_clkb));
     120       i_enb =1'b1;
     121       i_web =1'b0;
     122      // $display("Data Read from the memory is %0d",dut.o_doutb );
     123       repeat(5) @( posedge (i_clkb));
     124       i_enb =1'b0;
     125       i_web =1'b1;
     126       repeat(5) @( posedge (i_clkb));
     127       i_enb =1'b0;
     128       i_web =1'b0;
     129       repeat(5) @( posedge (i_clkb));
     130       i_enb =1'b1;
     131       i_web =1'b0;

     133       i_enb =1'b1;
     134       i_web =1'b1;
